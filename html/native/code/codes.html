const btnCodeCopy = document.getElementById("code-copy");
const blockOfCode = document.getElementById("code-block");

async function getCopyOfCode() {
await navigator.clipboard.writeText(blockOfCode.textContent);
btnCodeCopy.textContent = "Copied";

setTimeout(() => {
   btnCodeCopy.textContent = "Copy";
   }, 800);
}

function listener(node, event, handler) {
  node.addEventListener(event, handler);
  return () => node.removeEventListener(event, handler);
}

listener(btnCodeCopy, "click", getCopyOfCode);
